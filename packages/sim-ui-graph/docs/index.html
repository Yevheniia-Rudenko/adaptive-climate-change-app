@climateinteractive/sim-ui-graph #

This package makes it easier to incorporate graphs into a web application that is built around an SDEverywhere-generated model.

Example #

import { getDefaultConfig, createDefaultModel } from '@climateinteractive/en-roads-core'
import { GraphView } from '@climateinteractive/sim-ui-graph'

let model
let modelContext
let graphView

function createGraphViewModel(graphSpec) {
  const datasetViewModels = graphSpec.datasets.map(datasetSpec => {
    return {
      spec: datasetSpec,
      visible: true,
      points: []
    }
  })
  return {
    spec: graphSpec,
    getDatasets: () => datasetViewModels,
    getStringForKey: key => {
      // TODO: Inject values if string is templated
      return str(key)
    },
    formatYAxisTickValue: value => {
      // Can use d3-format here and pass graphSpec.yFormat
      return value.toString()
    },
    formatYAxisTooltipValue: value => {
      // Can use d3-format here and pass '.2~f', for example
      return value.toString()
    }
  }
}

function updateGraphData() {
  if (graphView) {
    const graphViewModel = graphView.viewModel
    for (const datasetViewModel of graphViewModel.getDatasets()) {
      const datasetSpec = datasetViewModel.spec
      const series = modelContext.getSeriesForVar(datasetSpec.varId, datasetSpec.externalSourceName)
      datasetViewModel.points = series?.points || []
    }
    graphView.updateData()
  }
}

function showGraph(graphSpec) {
  if (graphView) {
    // Destroy the old view before switching to a new one
    graphView.destroy()
  }

  const canvas = $('#graph-canvas')[0]
  const viewModel = createGraphViewModel(graphSpec)
  const style = {
    font: {
      family: 'Helvetica, sans-serif',
      style: 'bold',
      color: '#231f20'
    },
    xAxis: {
      tickMaxCount: 6
    },
    getAxisLabelFontSize: () => window.innerWidth * (1.0 / 100),
    getTickLabelFontSize: () => window.innerWidth * (1.2 / 100),
    getDefaultLineWidth: () => window.innerWidth * (0.5 / 100)
  }
  const options = {
    style
  }
  const tooltipsEnabled = true
  const xAxisLabel = str(graphSpec.xAxisLabel)
  const yAxisLabel = str(graphSpec.yAxisLabel)
  graphView = new GraphView(canvas, viewModel, options, tooltipsEnabled, xAxisLabel, yAxisLabel)
  updateGraphData()
}

async function initApp() {
  // Load the core En-ROADS configuration
  const config = getDefaultConfig()

  // Initialize the En-ROADS model asynchronously
  model = await createAsyncModel()
  modelContext = model.addContext()

  // Create a `GraphView` that is bound to an HTML element
  showGraph(config.graphs.get('1'))

  // When the model outputs are updated, refresh the graph
  modelContext.onOutputsChanged(() => {
    updateGraphData()
  })
}
